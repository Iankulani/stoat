# STOAT-HACKER v4.0.0 - Makefile
# Usage: make [target]

.PHONY: help install run test clean docker-build docker-run docker-stop deploy

# Colors
RED=\033[0;31m
GREEN=\033[0;32m
YELLOW=\033[1;33m
BLUE=\033[0;34m
NC=\033[0m

help:
	@echo "$(BLUE)🦡 STOAT-HACKER v4.0.0 - Make Commands$(NC)"
	@echo ""
	@echo "$(GREEN)Available targets:$(NC)"
	@echo "  $(YELLOW)install$(NC)    - Install dependencies and setup environment"
	@echo "  $(YELLOW)run$(NC)        - Run STOAT-HACKER"
	@echo "  $(YELLOW)test$(NC)       - Run test suite"
	@echo "  $(YELLOW)check$(NC)      - Check requirements"
	@echo "  $(YELLOW)clean$(NC)      - Clean temporary files"
	@echo "  $(YELLOW)docker-build$(NC) - Build Docker image"
	@echo "  $(YELLOW)docker-run$(NC)  - Run Docker container"
	@echo "  $(YELLOW)docker-stop$(NC) - Stop Docker container"
	@echo "  $(YELLOW)deploy$(NC)     - Deploy to production"
	@echo "  $(YELLOW)update$(NC)     - Update to latest version"
	@echo ""

install:
	@echo "$(BLUE)📦 Installing STOAT-HACKER...$(NC)"
	@chmod +x setup.sh
	@./setup.sh

run:
	@echo "$(BLUE)🚀 Running STOAT-HACKER...$(NC)"
	@./stoat.sh

test:
	@echo "$(BLUE)🧪 Running tests...$(NC)"
	@python3 test_commands.py

check:
	@echo "$(BLUE)🔍 Checking requirements...$(NC)"
	@python3 requirements_check.py

clean:
	@echo "$(BLUE)🧹 Cleaning temporary files...$(NC)"
	@rm -rf __pycache__ *.pyc *.pyo
	@rm -rf temp/* logs/*.log
	@rm -rf stoat_reports/*.txt stoat_reports/*.html
	@find . -name "*.pyc" -delete
	@find . -name "__pycache__" -delete
	@echo "$(GREEN)✅ Clean complete$(NC)"

docker-build:
	@echo "$(BLUE)🐳 Building Docker image...$(NC)"
	@docker build -t stoat-hacker:latest .

docker-run:
	@echo "$(BLUE)🐳 Running Docker container...$(NC)"
	@docker run -it --rm --net=host -v $(PWD)/.stoat:/app/.stoat -v $(PWD)/logs:/app/logs stoat-hacker:latest

docker-stop:
	@echo "$(BLUE)🐳 Stopping Docker container...$(NC)"
	@docker stop stoat-hacker || true
	@docker rm stoat-hacker || true

deploy:
	@echo "$(BLUE)🚀 Deploying to production...$(NC)"
	@sudo systemctl stop stoat || true
	@git pull
	@./setup.sh
	@sudo systemctl start stoat
	@sudo systemctl status stoat
	@echo "$(GREEN)✅ Deployed successfully$(NC)"

update:
	@echo "$(BLUE)🔄 Updating to latest version...$(NC)"
	@git pull
	@./setup.sh
	@echo "$(GREEN)✅ Update complete$(NC)"

# Create a backup
backup:
	@echo "$(BLUE)💾 Creating backup...$(NC)"
	@tar -czf stoat_backup_$(shell date +%Y%m%d_%H%M%S).tar.gz .stoat stoat_reports logs
	@echo "$(GREEN)✅ Backup complete$(NC)"

# Development environment
dev:
	@echo "$(BLUE)🛠️ Setting up development environment...$(NC)"
	@pip install -r requirements.txt -q
	@pip install -r requirements_dev.txt 2>/dev/null || true
	@echo "$(GREEN)✅ Dev environment ready$(NC)"

# Quick start
quick:
	@echo "$(BLUE)⚡ Quick start...$(NC)"
	@python3 stoat.py
